sequenceDiagram
    participant User as 用户
    participant InputBar as InputBar
    participant Bridge as tauri-bridge
    participant Rust as Rust Backend
    participant CLI as Claude CLI
    participant Protocol as protocol.rs
    participant SP as useStreamProcessor
    participant Chat as chat store
    participant Vue as Vue 组件

    User->>InputBar: 输入消息 + 文件
    InputBar->>Bridge: sendMessage(sessionId, text, options)
    Bridge->>Rust: invoke("send_message")
    Rust->>Rust: 保存用户消息到 SQLite
    Rust->>Rust: sync_permission_settings()
    Rust->>CLI: spawn claude CLI (stream-json)

    loop NDJSON 逐行
        CLI-->>Protocol: stdout line
        Protocol->>Protocol: StreamLine::parse(line)
        Protocol->>Protocol: to_frontend_event(sessionId)
        Protocol->>Rust: emit("stream-event", event)
        Rust-->>SP: Tauri event
        SP->>SP: 去重检查 (startsWith)
        SP->>Chat: appendText / appendThinking / addToolUse
        Chat-->>Vue: 响应式 re-render
    end

    CLI-->>Rust: 进程退出
    Rust->>SP: emit("process-exited")
    SP->>Chat: finishAssistantMessage(stats)
    SP->>SP: 桌面通知 notifyComplete()
